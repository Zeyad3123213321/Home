How to connect this form to your Google Sheet

  1. Create a new Google Sheet and note its ID from the URL
  2. Open Google Apps Script and create a new project
  3. Paste the following code into the script editor:
// Replace 'YOUR_SHEET_ID_HERE' with your actual Google Sheet ID
const SHEET_ID = 'YOUR_SHEET_ID_HERE';

function doPost(e) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
    const data = JSON.parse(e.postData.contents);
    
    // Add headers if the sheet is empty
    if (sheet.getLastRow() === 0) {
      sheet.getRange(1, 1, 1, 5).setValues([['Timestamp', 'Name', 'Email', 'Phone', 'Department', 'Message']]);
    }
    
    // Add new row with data
    sheet.appendRow([
      new Date(),
      data.name,
      data.email,
      data.phone,
      data.department,
      data.message
    ]);
    
    return ContentService
      .createTextOutput(JSON.stringify({result: 'success'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({result: 'error', error: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  return ContentService.createTextOutput("This web app is for receiving form data.");
}

    • Execute as: Me
